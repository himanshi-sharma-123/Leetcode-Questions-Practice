class Solution {
    public int[] countBits(int n) {
        int ans[] = new int[n+1];
        for(int i=0;i<=n;i++){
            int count = solve(i);
            ans[i] = count;
        }

        return ans;
    }

    public int solve(int n){
        int cnt = 0;
        while(n!=0){
            cnt += n&1;
            n >>>= 1;
        }

        return cnt;
    }
}
